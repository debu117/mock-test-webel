app.post('/api/submit-test', async (req, res) => {
    const { userId, studentName, answers } = req.body; // 'answers' is an array: [{questionId, selectedOption}]
    let totalCorrect = 0;
    let reportDetails = [];

    for (let item of answers) {
        // 1. Fetch the true correct answer from your secure database
        const question = await db.getQuestion(item.questionId); 
        const isCorrect = (item.selectedOption === question.correctOption);

        if (isCorrect) totalCorrect++;

        // 2. Build the detailed breakdown for the admin
        reportDetails.push({
            questionText: question.text,
            studentAnswer: item.selectedOption,
            correctAnswer: question.correctOption,
            status: isCorrect ? "Correct" : "Wrong"
        });
    }

    // 3. Save the full results into the Test_Submissions database
    await db.saveSubmission({
        userId,
        studentName,
        score: totalCorrect,
        breakdown: reportDetails
    });

    res.status(200).json({ message: "Test submitted successfully!" });
});
